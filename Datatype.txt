Make your own datatype drank
CREATE TYPE drank AS (
    merk VARCHAR(200),
    soort VARCHAR(200),
    brouwerij VARCHAR(255),
    periode VARCHAR(255)
);

CREATE TABLE drank_tabel (
    merknaam drank NOT NULL UNIQUE
);
